// 实时字数统计
const issueDesc = document.getElementById('issueDesc');
const currentCount = document.getElementById('currentCount');
const charCount = document.querySelector('.char-count');

issueDesc.addEventListener('input', function() {
    const count = this.value.length;
    const max = 500;
    
    currentCount.textContent = count;
    
    // 字数预警

        charCount.classList.add('danger');
        charCount.classList.remove('warning');

        charCount.classList.add('warning');
        charCount.classList.remove('danger');
    } else {
        charCount.classList.remove('warning');
        charCount.classList.remove('danger');
    }
});

function validateIssueDesc() {
    const desc = document.getElementById('issueDesc').value.trim();
    const max = 500;

    }
    
    return true;
}

// 初始化字数统计
function initCharCount() {
    const issueDesc = document.getElementById('issueDesc');
    const currentCount = document.getElementById('currentCount');
    const charCount = document.getElementById('charCount');
    const max = 500;
    const warning = 400;
    
    issueDesc.addEventListener('input', function() {
        const count = this.value.length;
        currentCount.textContent = count;
        
        // 移除所有状态类
        charCount.classList.remove('warning', 'danger');
        
        // 添加状态类
        if (count >= max) {
            charCount.classList.add('danger');
        } else if (count >= warning) {
            charCount.classList.add('warning');
        }
    });
}

// 验证问题描述
function validateIssueDesc() {
    const desc = document.getElementById('issueDesc').value.trim();
    const max = 500;
    const min = 10;
    
    if (!desc) {
        showToast('请填写问题描述（至少 10 字）');
        return false;
    }
    
    if (desc.length < min) {
        showToast(`问题描述太短，请至少输入${min}字（当前${desc.length}字）`);
        return false;
    }
    
    if (desc.length > max) {
        showToast(`问题描述超长，请精简到${max}字以内（当前${desc.length}字）`);
        return false;
    }
    
    return true;
}

// 页面加载时初始化
document.addEventListener('DOMContentLoaded', function() {
    initCharCount();
});
