```javascript
let selectedTypes = [];

function toggleType(element) {
    element.classList.toggle('selected');
    const type = element.dataset.type;
    if (selectedTypes.includes(type)) {
        selectedTypes = selectedTypes.filter(t => t !== type);
    } else {
        selectedTypes.push(type);
    }
}

    // ...
};
```
